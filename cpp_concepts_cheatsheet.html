ARRAYS — STACK & HEAP
// Stack arrays (fixed size, fast)
int arr[5];               // uninitialized
int arr2[5] = {};         // all zeros
int arr3[5] = {1,2,3};   // {1,2,3,0,0}
int arr4[] = {1,2,3};    // size deduced = 3

for(int i=0;i<3;i++) delete[] mat[i];

CLASSES — ALL INIT STYLES
class MyClass {
public:
    int val;
    std::string name;
    static int count;     // shared across all instances

    // Default constructor
    MyClass() : val(0), name("default") {}

    // Parameterized
    MyClass(int v, std::string n)
        : val(v), name(std::move(n)) {}

    // Copy constructor
    MyClass(const MyClass& o)
        : val(o.val), name(o.name) {}

    // Move constructor (C++11)
    MyClass(MyClass&& o) noexcept
        : val(o.val), name(std::move(o.name)) {}
};

// Instantiation styles
MyClass a;                 // default ctor
MyClass b(5, "hi");        // direct init
MyClass c = MyClass(5,"x"); // copy init
MyClass d{5, "hi"};        // list init (C++11)
MyClass* e = new MyClass(5, "hi");
delete e;
auto f = MyClass(5, "hi");  // type deduced

RULE OF 3/5/0
class Buffer {
    int* data;
    int size;
public:
    Buffer(int n) : data(new int[n]), size(n) {}

    // Destructor
    ~Buffer() { delete[] data; }

    // Copy constructor (deep copy)
    Buffer(const Buffer& o)
        : data(new int[o.size]), size(o.size) {
        std::copy(o.data, o.data+o.size, data);
    }

    // Copy assignment
    Buffer& operator=(const Buffer& o) {
        if(this == &o) return *this;
        delete[] data;
        data = new int[o.size];
        size = o.size;
        std::copy(o.data, o.data+o.size, data);
        return *this;
    }

    // Move constructor (C++11, steal resources)
    Buffer(Buffer&& o) noexcept
        : data(o.data), size(o.size)
    { o.data = nullptr; o.size = 0; }

    // Move assignment
    Buffer& operator=(Buffer&& o) noexcept {
        if(this == &o) return *this;
        delete[] data;
        data = o.data; size = o.size;
        o.data = nullptr; o.size = 0;
        return *this;
    }
};
// Rule of 0: use smart ptrs — no manual copy/move needed
STACK vs HEAP SUMMARY
// ── STACK ───────────────────────────────
// Automatic lifetime (freed when scope ends)
// Fast allocation (just move stack pointer)
// Size must be known at compile time
{
    int x = 5;           // lives on stack
    int arr[100];        // stack array, fixed
}   // x, arr automatically freed here

// ── HEAP ────────────────────────────────
// Manual lifetime (new/delete)
// Slower allocation (OS involved)
// Dynamic size possible
int* p = new int(5);    // heap, stays until deleted
delete p;               // must free manually

// ── COMMON PITFALLS ──────────────────────
// Memory leak:  new without delete
// Double free:  delete called twice
// Dangling ptr: use after delete
// Buffer overrun: write past allocated size
// Use smart pointers to avoid all of these!
MEMORY UTILITIES
// memset / memcpy / memmove
int a[5];
memset(a, 0, sizeof(a));       // fill bytes with 0
memset(a, -1, sizeof(a));      // fill with 0xFF (all bits 1)
memcpy(dst, src, 5*sizeof(int));// copy 20 bytes (no overlap)
memmove(dst,src,5*sizeof(int));// safe even if overlap

// sizeof / alignof
sizeof(int)           // 4 bytes
sizeof(long long)     // 8 bytes
sizeof(arr)/sizeof(arr[0])  // array length
alignof(double)       // alignment requirement = 8

// std::fill / std::memset equivalents
std::fill(a, a+5, 0);           // preferred for non-POD
std::fill_n(a, 5, 42);

// Aligned allocation (C++17)
void* p = std::aligned_alloc(64, 512);
std::free(p);
⚠ Always pair new↔delete and new[]↔delete[]. Mixing them is undefined behavior. Prefer smart pointers (below) to eliminate manual memory management entirely.

Smart Pointers

#include <memory> · unique_ptr · shared_ptr · weak_ptr
unique_ptr — EXCLUSIVE OWNERSHIP
// Creation — prefer make_unique (C++14)
auto p1 = std::make_unique<int>(42);
auto p2 = std::make_unique<Node>(5);
auto p3 = std::make_unique<int[]>(10); // array

// From raw pointer (avoid if possible)
std::unique_ptr<int> p4(new int(10));

// Access
*p1                    // dereference: 42
p2->val                // member access
p1.get()               // raw pointer (don't delete!)
p3[0]                  // array element

// Ownership transfer (move-only)
auto p5 = std::move(p1); // p1 is now null
someFunc(std::move(p5)); // transfer to function

// Release / reset
int* raw = p1.release(); // give up ownership
p2.reset();             // destroy held object
p2.reset(new Node(9)); // destroy old, own new

// Null check
if (p1) { /* not null */ }
if (p1 != nullptr) { /* same */ }

// Custom deleter
auto del = [](int* p){ /* custom cleanup */ delete p; };
std::unique_ptr<int, decltype(del)> p6(new int, del);
shared_ptr — SHARED OWNERSHIP
// Creation — prefer make_shared (single allocation)
auto s1 = std::make_shared<int>(42);
auto s2 = std::make_shared<Node>(5);

// From raw pointer
std::shared_ptr<int> s3(new int(10));

// Copy — both own the object
auto s4 = s1;            // ref count = 2
std::shared_ptr<int> s5 = s1;

// Access (same as unique_ptr)
*s1                      // dereference
s2->val                  // member
s1.get()                 // raw pointer

// Reference counting
s1.use_count()           // number of owners
s1.unique()              // true if only owner

// Reset
s1.reset();              // release ownership
s1.reset(new int(99));  // own new object

// Custom deleter
std::shared_ptr<FILE> fp(
    fopen("f.txt","r"), fclose);
// fclose called automatically!

// Aliasing constructor
std::shared_ptr<int> alias(s2, &s2->val);
weak_ptr — BREAK CYCLES
// weak_ptr doesn't own — doesn't affect ref count
// Use to break shared_ptr circular references

auto s = std::make_shared<int>(42);
std::weak_ptr<int> w = s;  // ref count still 1

// Check if object is alive
w.expired()              // true if object is gone
w.use_count()            // shared owners (not weak)

// Safely access object — lock() returns shared_ptr
if (auto locked = w.lock()) {
    cout << *locked;       // safe: object alive
} else {
    // object was destroyed
}

// Classic circular reference problem
struct NodeA {
    std::shared_ptr<NodeB> b;  // shared → cycle!
};
struct NodeA {
    std::weak_ptr<NodeB> b;    // weak → no cycle
};
SMART PTR PATTERNS & CASTING
// Passing smart pointers to functions
void take(std::unique_ptr<int> p);    // takes ownership
void use(const std::unique_ptr<int>& p); // borrow, no transfer
void peek(int* p);                    // pass raw with .get()

// Returning from factory functions
std::unique_ptr<Base> make() {
    return std::make_unique<Derived>();
}

// Casting shared_ptr
std::shared_ptr<Base> base = std::make_shared<Derived>();
auto derived = std::static_pointer_cast<Derived>(base);
auto derived2 = std::dynamic_pointer_cast<Derived>(base);
auto c = std::const_pointer_cast<Base>(constBase);

// unique_ptr in containers
std::vector<std::unique_ptr<Node>> nodes;
nodes.push_back(std::make_unique<Node>(1));
nodes.emplace_back(new Node(2));

// enable_shared_from_this
struct Foo : std::enable_shared_from_this<Foo> {
    std::shared_ptr<Foo> self() {
        return shared_from_this();
    }
};
Quick rule: Use unique_ptr by default. Use shared_ptr only when multiple owners are truly needed. Use weak_ptr to break cycles. Never call delete on a raw pointer that a smart pointer also manages.

Casting

static_cast · dynamic_cast · const_cast · reinterpret_cast · Numeric · String conversions
static_cast
// Safe compile-time cast between related types
// No runtime check — your responsibility!

// Numeric conversions
double d = 3.14;
int i = static_cast<int>(d);    // 3 (truncates)
long long ll = static_cast<long long>(42);

// Prevent integer overflow before multiply
long long res = static_cast<long long>(a) * b;

// Enum to int and back
enum class Color { Red, Green, Blue };
int n = static_cast<int>(Color::Green);  // 1
Color c = static_cast<Color>(2);          // Blue

// Pointer upcast (always safe)
Derived* d2 = new Derived();
Base* b = static_cast<Base*>(d2);

// Pointer downcast (unsafe — use dynamic_cast!)
Derived* d3 = static_cast<Derived*>(b);  // risky

// void* to typed pointer
void* vp = &i;
int* ip = static_cast<int*>(vp);
dynamic_cast · const_cast · reinterpret_cast
// dynamic_cast — safe downcast with runtime check
// Requires polymorphic base (virtual function)
Base* b = new Derived();
Derived* d = dynamic_cast<Derived*>(b);
if (d) { /* cast succeeded */ }
// Returns nullptr on failure (pointer)
// Throws std::bad_cast on failure (reference)
Derived& dr = dynamic_cast<Derived&>(*b);

// const_cast — add or remove const
const int* cp = &i;
int* mp = const_cast<int*>(cp);   // remove const
// Only valid if underlying object is non-const!

void print(char* s);
const char* cs = "hello";
print(const_cast<char*>(cs));    // legacy API

// reinterpret_cast — raw bit reinterpretation
// Dangerous! Use only when you know exactly what you're doing.
int x = 65;
char* cp2 = reinterpret_cast<char*>(&x);
// Use case: serialization, hardware registers

// Type punning (C++20 use std::bit_cast)
float f = 1.0f;
uint32_t bits = std::bit_cast<uint32_t>(f); // safe

Lambda Functions

C++11+ · [capture](params) → return { body }
CAPTURE MODES
int x = 10, y = 20;
std::vector<int> v = {1,2,3};

// [] — capture nothing
auto f1 = []{ return 42; };

// [=] — capture all by value (copy)
auto f2 = [=]{ return x + y; };   // x,y copied

// [&] — capture all by reference
auto f3 = [&]{ x++; y++; };       // modifies originals

// Selective capture
auto f4 = [x]{ return x; };       // only x by value
auto f5 = [&x]{ x++; };           // only x by reference
auto f6 = [x, &y]{ y += x; };    // x by val, y by ref
auto f7 = [=, &v]{ v.push_back(x); }; // default val, v by ref
auto f8 = [&, x]{ y += x; };     // default ref, x by val

// Capture with init (C++14)
auto f9 = [z = x + y]{ return z; };       // new var z
auto f10 = [p = std::move(v)]() mutable { // move into lambda
    p.push_back(9);
};

// this capture (in class method)
auto f11 = [this]{ return this->val; };
auto f12 = [*this]{ return val; };  // C++17, copies this
PARAMETERS & RETURN TYPES
// No params
auto f1 = []{ return 42; };
auto f2 = []() { return 42; };   // same

// With params
auto add = [](int a, int b) { return a + b; };
add(3, 4);                        // 7

// Explicit return type
auto div = [](int a, int b) -> double {
    return (double)a / b;
};

// Generic lambda (C++14) — auto parameters
auto mymax = [](auto a, auto b) { return a > b ? a : b; };
mymax(3, 4);           // works for int, double, etc.
mymax("abc", "xyz");

// Variadic generic lambda (C++14)
auto print = [](auto&&... args) {
    ((cout << args << ' '), ...);
};

// mutable — allows modifying captured-by-value
int cnt = 0;
auto counter = [cnt]() mutable { return ++cnt; };
// original cnt is unchanged; lambda has its own copy

// noexcept lambda
auto safe = [](int x) noexcept { return x * 2; };
STORING & PASSING LAMBDAS
// Store in auto (simple)
auto f = [](int x) { return x * 2; };

// Store in std::function (type-erased, slower)
std::function<int(int)> g = [](int x){ return x*2; };
std::function<void()> h = [&]{ x++; };

// Pass to algorithms
sort(v.begin(), v.end(), [](int a, int b){ return a>b; });
find_if(v.begin(), v.end(), [](int x){ return x>3; });
transform(v.begin(), v.end(), v.begin(), [](int x){ return x*2; });
count_if(v.begin(), v.end(), [](int x){ return x%2==0; });

// Accept as template parameter (fastest)
template<typename F>
void apply(std::vector<int>& v, F fn) {
    for(auto& x : v) x = fn(x);
}
apply(v, [](int x){ return x*3; });
RECURSIVE LAMBDAS & IIFE
// IIFE — Immediately Invoked Function Expression
int result = [](int n){ return n*n; }(5);  // 25

// IIFE for complex initialization
const auto data = []() {
    std::vector<int> v;
    for(int i=0;i<10;i++) v.push_back(i*i);
    return v;
}();

// Recursive lambda — use std::function
std::function<int(int)> fib = [&fib](int n) {
    return n <= 1 ? n : fib(n-1) + fib(n-2);
};

// Recursive lambda — C++23 deducing this
auto fib2 = [](this auto self, int n) {
    return n <= 1 ? n : self(n-1) + self(n-2);
};

// DFS as recursive lambda (LeetCode pattern)
vector<int> res;
std::function<void(int, int)> dfs;
dfs = [&](int node, int parent) {
    res.push_back(node);
    for(int nb : adj[node])
        if(nb != parent) dfs(nb, node);
};
dfs(0, -1);
LAMBDAS AS COMPARATORS
// Custom sort — 2D point by distance
sort(pts.begin(), pts.end(), [](auto& a, auto& b){
    return a[0]*a[0]+a[1]*a[1] < b[0]*b[0]+b[1]*b[1];
});

// Custom priority_queue comparator
auto cmp = [](auto& a, auto& b){ return a.freq < b.freq; };
priority_queue<Item, vector<Item>, decltype(cmp)> pq(cmp);

// Custom map comparator
auto cmp2 = [](int a, int b){ return abs(a) < abs(b); };
map<int, int, decltype(cmp2)> m(cmp2);

// Sort strings by length then lexicographic
sort(words.begin(), words.end(), [](auto& a, auto& b){
    return a.size() != b.size() ? a.size() < b.size() : a < b;
});

String Parsing

Split · Tokenize · Regex · sscanf · Manual parsing
SPLIT / TOKENIZE
// Method 1: istringstream (split by whitespace)
std::string s = "hello world foo";
std::istringstream iss(s);
std::vector<std::string> tokens;
std::string tok;
while (iss >> tok) tokens.push_back(tok);

// Method 2: getline with delimiter
std::string csv = "a,b,c,d";
std::istringstream ss(csv);
while (std::getline(ss, tok, ',')) tokens.push_back(tok);

// Method 3: find + substr loop
std::string str = "one::two::three";
std::string delim = "::";
size_t pos = 0, found;
while ((found = str.find(delim, pos)) != std::string::npos) {
    tokens.push_back(str.substr(pos, found - pos));
    pos = found + delim.size();
}
tokens.push_back(str.substr(pos));  // last token

// Method 4: strtok (C-style, modifies string!)
char buf[] = "a,b,c";
char* p = strtok(buf, ",");
while (p) { tokens.push_back(p); p = strtok(nullptr, ","); }

// C++20 ranges split (conceptual)
// auto parts = str | views::split(',');
SEARCH & REPLACE IN STRINGS

// Replace all occurrences
std::string replaceAll(std::string s,
    const std::string& from,
    const std::string& to) {
    size_t pos = 0;
    while ((pos = s.find(from, pos)) != std::string::npos) {
        s.replace(pos, from.size(), to);
        pos += to.size();
    }
    return s;
}

MODULAR ARITHMETIC
const long long MOD = 1e9 + 7;

// Always mod intermediate results
long long res = (a + b) % MOD;
long long res2 = (a % MOD + b % MOD) % MOD;
long long prod = (a % MOD) * (b % MOD) % MOD;

// Subtraction mod (prevent negative)
long long diff = ((a - b) % MOD + MOD) % MOD;

// Fast power (modular exponentiation)
ll power(ll base, ll exp, ll mod) {
    ll result = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) result = result * base % mod;
        base = base * base % mod;
        exp >>= 1;
    }
    return result;
}

// Modular inverse (a^(MOD-2) % MOD, when MOD is prime)
ll modinv(ll a, ll mod) { return power(a, mod-2, mod); }
// Division under mod: (a/b) % MOD = a * modinv(b, MOD) % MOD

Critical reminders:
· new T[] must be paired with delete[], never plain delete.   · unique_ptr cannot be copied — only moved. shared_ptr has overhead from ref-counting — don't use it everywhere.   · dynamic_cast requires at least one virtual function in the base class; otherwise it won't compile.   · stoi/stod throw std::invalid_argument on bad input — wrap in try-catch for untrusted input.   · Lambda capture by reference [&] is dangerous if the lambda outlives the captured variable (e.g. in async tasks).   · std::function has type-erasure overhead — use auto for local lambdas, std::function only when you need to store or return them.   · Integer division truncates toward zero in C++: -7/2 == -3, not -4.   · Use 1LL * a * b to force 64-bit multiplication even when a and b are int.

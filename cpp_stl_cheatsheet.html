vector<T>

#include <vector> Random access O(1) · push_back amortized O(1) · insert/erase O(n)
INIT
vector<int> v1;                      // empty
vector<int> v2(5);                   // {0,0,0,0,0}
vector<int> v3(5, 42);              // {42,42,42,42,42}
vector<int> v4 = {1,2,3,4};        // initializer list
vector<int> v5(v4);                  // copy constructor
vector<int> v6(v4.begin(), v4.end()); // from iterators
vector<vector<int>> v7(3, vector<int>(4, 0)); // 3x4 grid of 0s
vector<int> v8(arr, arr + n);        // from C-array
INSERT / PUSH
v.push_back(10);                     // append to end O(1)
v.emplace_back(10);                  // construct in-place at end O(1)
v.insert(v.begin(), 5);             // insert at front O(n)
v.insert(v.begin() + 2, 5);        // insert at index 2
v.insert(v.end(), 3, 99);           // insert 3 copies of 99
v.insert(v.end(), v2.begin(), v2.end()); // append vector
v.assign(5, 0);                      // replace with 5 zeros
v.assign({1,2,3});                  // replace with list
v.resize(10);                        // grow (fill 0) or shrink
v.resize(10, -1);                   // grow with fill value -1
ERASE / POP
v.pop_back();                       // remove last O(1)
v.erase(v.begin());                 // remove first O(n)
v.erase(v.begin() + 2);            // remove at index 2
v.erase(v.begin(), v.begin() + 3); // remove first 3
v.clear();                          // remove all O(n)
// Erase-remove idiom (remove by value):
v.erase(remove(v.begin(),v.end(),5), v.end());
// Erase-remove with predicate:
v.erase(remove_if(v.begin(),v.end(),[](
  int x){return x%2==0;}),v.end());
ACCESS
v[2]                               // index (no bounds check)
v.at(2)                            // index (throws if OOB)
v.front()                          // first element
v.back()                           // last element
v.data()                           // raw pointer to array
ITERATE
// Index-based:
for (int i = 0; i < v.size(); i++) cout << v[i];

// Range-based (read):
for (auto x : v) cout << x;

// Range-based (modify):
for (auto& x : v) x *= 2;

// Iterator:
for (auto it = v.begin(); it != v.end(); it++) cout << *it;

// Reverse:
for (auto it = v.rbegin(); it != v.rend(); it++) cout << *it;
SEARCH
// Linear search (unsorted):
auto it = find(v.begin(), v.end(), 5);
if (it != v.end()) // found at index: it-v.begin()

// Binary search (MUST be sorted):
binary_search(v.begin(), v.end(), 5); // true/false
lower_bound(v.begin(), v.end(), 5);  // iter to first >= 5
upper_bound(v.begin(), v.end(), 5);  // iter to first >  5

// Count occurrences:
count(v.begin(), v.end(), 5);
count_if(v.begin(), v.end(), [](int x){return x>3;});
SORT / REVERSE
sort(v.begin(), v.end());           // ascending O(n log n)
sort(v.begin(), v.end(), greater<int>()); // descending
sort(v.begin(), v.end(), [](int a, int b){return a<b;});
stable_sort(v.begin(), v.end());    // preserves equal order
reverse(v.begin(), v.end());        // in-place reverse
nth_element(v.begin(),v.begin()+k,v.end()); // kth element
partial_sort(v.begin(),v.begin()+k,v.end()); // first k sorted
MISC
v.size()                           // number of elements
v.empty()                          // true if empty
v.capacity()                       // allocated memory
v.reserve(100)                     // pre-allocate capacity
v.shrink_to_fit()                  // release excess memory
swap(v1, v2)                       // swap two vectors O(1)
min_element(v.begin(), v.end())    // iterator to min
max_element(v.begin(), v.end())    // iterator to max
accumulate(v.begin(), v.end(), 0) // sum (needs <numeric>)
fill(v.begin(), v.end(), 0)       // set all to 0
iota(v.begin(), v.end(), 1)       // fill 1,2,3,... (<numeric>)
unique(v.begin(), v.end())        // remove consec dups (sort first!)

for (int i = 0; i < s.size(); i++) cout << s[i];

for (auto it = s.begin(); it != s.end(); it++) cout << *it;

for (auto it = s.rbegin(); it != s.rend(); it++) cout << *it; // reverse

s.find_first_of("aeiou")           // first vowel pos
s.find_last_of("aeiou")            // last vowel pos

// Check if found:
if (s.find("lo") != string::npos) { /* found */ }
SORT / REVERSE

sort(s.begin(), s.end(), greater<char>());
reverse(s.begin(), s.end());        // reverse in-place
next_permutation(s.begin(), s.end()); // next lex permutation
MISC
s.size()  / s.length()             // number of chars
s.empty()                          // true if empty
s1 + s2                            // concatenation (new string)
s1 == s2 / s1 < s2               // comparison (lexicographic)
s.compare(s2)                      // 0 if equal
tolower(s[i]) / toupper(s[i])    // char conversion
isalpha(c) / isdigit(c) / isspace(c)

list<T>  (doubly linked)

#include <list> Insert/delete at iterator O(1) · No random access · LRU cache use case
INIT
list<int> l1;                       // empty
list<int> l2 = {1,2,3};
list<int> l3(5, 0);               // 5 zeros
list<int> l4(v.begin(), v.end()); // from vector
INSERT
l.push_front(1);                   // prepend O(1)
l.push_back(5);                    // append O(1)
l.emplace_front(1);               // construct in-place front
l.emplace_back(5);                // construct in-place back
auto it = l.begin(); advance(it,2);
l.insert(it, 99);                  // insert before iterator O(1)
l.insert(it, 3, 99);              // insert 3 copies
l.splice(it, l2);                  // move all of l2 before it
l.splice(it, l2, l2.begin());     // move one element
ERASE
l.pop_front();                     // remove first O(1)
l.pop_back();                      // remove last O(1)
l.erase(it);                       // remove at iterator O(1)
l.erase(it1, it2);                 // remove range
l.remove(5);                       // remove all elements == 5
l.remove_if([](int x){return x<0;}); // remove by predicate
l.unique();                        // remove consecutive dups
l.clear();
ACCESS ITERATE SORT
l.front() / l.back()               // no random access!
l.size() / l.empty()

for (auto x : l) cout << x;
for (auto it = l.begin(); it != l.end(); ++it) cout << *it;

l.sort();                          // list has its own sort O(n log n)
l.sort(greater<int>());
l.reverse();                       // list has its own reverse O(n)
l.merge(l2);                       // merge two sorted lists

  vector<pair<int,int>>,

    return a.second > b.second; // min on second
};

  vector<pair<int,int>>,
  decltype(cmp)> custpq(cmp);

pq.pop();                          // remove top O(log n) — void!

pq.size() / pq.empty()

Quick tips:
· Always use long long for sums that may exceed 2³¹ (~2.1B).   · unordered_* containers can degrade to O(n) with hash collisions — use reserve() and custom hash for large inputs.   · mp[key] on an unordered_map/map silently inserts a default value if key is missing — use count() or find() to check first.   · multiset::erase(val) removes ALL copies — use erase(find(val)) to remove exactly one.   · stack::pop() and queue::pop() return void — always call top()/front() first.   · unique() only removes consecutive duplicates — sort first if you want full dedup.   · Include #include <bits/stdc++.h> to get everything in one line (GCC only).
